
package graphical.test;

import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;


public class FormController implements ActionListener{
    
 private   MainFrame mainFrm;
    
    public FormController (){
        
       mainFrm = new MainFrame();
        bindActionListener();
        mainFrm.setVisible(true);
    }
    private void bindActionListener(){
        
        mainFrm.getExitBtn().addActionListener(this);
         mainFrm.getTextBtn().addActionListener(this);
    }
    
    private void setLabelText(){
        
        mainFrm.setUserLabel("Én vagyok a Jani");
         mainFrm.setStatusLabel("Cseréltem a szöveget");
        
    }
    private void exit() {
        System.exit(0);
    }

    
    public void actionPerformed(ActionEvent event){
        
        
        if(event.getSource() == mainFrm.getTextBtn()){
            
            setStatusText();
        }else if(event.getSource() == mainFrm.getExitBtn()){
            exit();
        }
        
    
}
